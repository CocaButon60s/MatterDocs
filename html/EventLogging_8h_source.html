69template <typename T>
70CHIP_ERROR LogEvent(const T & aEventData, EndpointId aEndpoint, EventNumber & aEventNumber)
71{
72 EventLogger<T> eventData(aEventData);
73
74 EventOptions eventOptions;
75 eventOptions.mPath = ConcreteEventPath(aEndpoint, aEventData.GetClusterId(), aEventData.GetEventId());
76 eventOptions.mPriority = aEventData.GetPriorityLevel();
77
78 if constexpr (DataModel::IsFabricScoped<T>::value)
79 {
80 eventOptions.mFabricIndex = aEventData.GetFabricIndex();
81 // A fabric-sensitive event must be associated with a fabric to make sense.
82 VerifyOrReturnError(eventOptions.mFabricIndex != kUndefinedFabricIndex, CHIP_ERROR_INVALID_FABRIC_INDEX);
83 }
84
85 return EventManagement::GetInstance().LogEvent(&eventData, eventOptions, aEventNumber);
86}
