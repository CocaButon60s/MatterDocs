62// The attrMask must contain the relevant MATTER_ATTRIBUTE_FLAG_* bits from
63// attribute-metadata.h. Specifically:
64//
65// * Writable attributes must have MATTER_ATTRIBUTE_FLAG_WRITABLE
66// * Nullable attributes (have X in the quality column in the spec) must have MATTER_ATTRIBUTE_FLAG_NULLABLE
67// * Attributes that have T in the Access column in the spec must have MATTER_ATTRIBUTE_FLAG_MUST_USE_TIMED_WRITE
68//
69// NOTE: ZAP_ATTRIBUTE_MASK(READABLE) is added by default to ensure backward compatibility, since DECLARE_DYNAMIC_ATTRIBUTE() is a
70// widely used API. If you want to add a write-only dynamic attribute, either expand the macro or contribute a
71// DECLARE_WRITEONLY_DYNAMIC_ATTRIBUTE API.
72#define DECLARE_DYNAMIC_ATTRIBUTE(attId, attType, attSizeBytes, attrMask) \
73 { \
74 ZAP_EMPTY_DEFAULT(), attId, attSizeBytes, ZAP_TYPE(attType), \
75 attrMask | ZAP_ATTRIBUTE_MASK(EXTERNAL_STORAGE) | ZAP_ATTRIBUTE_MASK(READABLE) \
76 }

246// Register a dynamic endpoint. This involves registering descriptors that describe
247// the composition of the endpoint (encapsulated in the 'ep' argument) as well as providing
248// storage for data versions.
249//
250// dataVersionStorage.size() needs to be at least as large as the number of
251// server clusters on this endpoint. If it's not, the endpoint will not be able
252// to store data versions, which may break consumers.
253//
254// The memory backing dataVersionStorage needs to remain allocated until this dynamic
255// endpoint is cleared.
256//
257// An optional device type list can be passed in as well. If provided, the memory
258// backing the list needs to remain allocated until this dynamic endpoint is cleared.
259//
260// An optional parent endpoint id should be passed for child endpoints of composed device.
261//
262// Returns CHIP_NO_ERROR No error.
263// CHIP_ERROR_NO_MEMORY MAX_ENDPOINT_COUNT is reached or when no storage is left for clusters
