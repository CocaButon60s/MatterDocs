47template <class Impl>
48class ScopedMemoryBufferBase
49{
50public:
51 ScopedMemoryBufferBase() {}
52 ScopedMemoryBufferBase(const ScopedMemoryBufferBase &) = delete;
53 ScopedMemoryBufferBase & operator=(const ScopedMemoryBufferBase & other) = delete;
54
55 ScopedMemoryBufferBase(ScopedMemoryBufferBase && other) { *this = std::move(other); }
56
57 ScopedMemoryBufferBase & operator=(ScopedMemoryBufferBase && other)
58 {
59 if (this != &other)
60 {
61 mBuffer = other.mBuffer;
62 other.mBuffer = nullptr;
63 }
64 return *this;
65 }
66
67 ~ScopedMemoryBufferBase() { Free(); }
68
70 explicit operator bool() const { return mBuffer != nullptr; }
71 bool operator!() const { return mBuffer == nullptr; }
72 inline bool IsNull() const { return mBuffer == nullptr; }
73
75 void Free()
76 {
77 if (mBuffer == nullptr)
78 {
79 return;
80 }
81 Impl::MemoryFree(mBuffer);
82 mBuffer = nullptr;
83 }
84
85protected:
86 void * Ptr() { return mBuffer; }
87 const void * Ptr() const { return mBuffer; }
88
94 CHECK_RETURN_VALUE void * Release()
95 {
96 void * buffer = mBuffer;
97 mBuffer = nullptr;
98 return buffer;
99 }
100
101 void Alloc(size_t size)
102 {
103 Free();
104 mBuffer = Impl::MemoryAlloc(size);
105 }
106
107 void Calloc(size_t elementCount, size_t elementSize)
108 {
109 Free();
110 mBuffer = Impl::MemoryCalloc(elementCount, elementSize);
111 }
112
113private:
114 void * mBuffer = nullptr;
115};

138template <typename T, class MemoryManagement = Impl::PlatformMemoryManagement>
139class ScopedMemoryBuffer : public Impl::ScopedMemoryBufferBase<MemoryManagement>
140{
141 friend class Impl::ScopedMemoryBufferBase<MemoryManagement>;
142
143public:
144 using Base = Impl::ScopedMemoryBufferBase<MemoryManagement>;
145
146 static_assert(std::is_trivially_destructible<T>::value, "Destructors won't get run");
147
148 T * Get() { return static_cast<T *>(Base::Ptr()); }
149 T & operator[](size_t index) { return Get()[index]; }
150
151 const T * Get() const { return static_cast<const T *>(Base::Ptr()); }
152 const T & operator[](size_t index) const { return Get()[index]; }
153
159 CHECK_RETURN_VALUE T * Release() { return static_cast<T *>(Base::Release()); }
160
161 ScopedMemoryBuffer & Calloc(size_t elementCount)
162 {
163 Base::Calloc(elementCount, sizeof(T));
164 ExecuteConstructors(elementCount);
165 return *this;
166 }
167
168 ScopedMemoryBuffer & Alloc(size_t elementCount)
169 {
170 Base::Alloc(elementCount * sizeof(T));
171 ExecuteConstructors(elementCount);
172 return *this;
173 }
174
175private:
176 template <typename U = T, std::enable_if_t<std::is_trivial<U>::value, int> = 0>
177 void ExecuteConstructors(size_t elementCount)
178 {
179 // Do nothing if our type is trivial. In particular, if we are a buffer
180 // of integers, we should not go zero-initializing them here: either
181 // caller wants that and called Calloc(), or it doesn't and we shouldn't
182 // do it.
183 }
184
185 template <typename U = T, std::enable_if_t<!std::is_trivial<U>::value, int> = 0>
186 void ExecuteConstructors(size_t elementCount)
187 {
188 T * elementPtr = Get();
189 if (elementPtr == nullptr)
190 {
191 // Alloc failed, don't bother.
192 return;
193 }
194 for (size_t i = 0; i < elementCount; ++i)
195 {
196 new (&elementPtr[i]) T();
197 }
198 }
199};

207template <typename T>
208class ScopedMemoryBufferWithSize : public ScopedMemoryBuffer<T>
209{
210public:
211 ScopedMemoryBufferWithSize() {}
212 ScopedMemoryBufferWithSize(ScopedMemoryBufferWithSize && other) { *this = std::move(other); }
213
214 ScopedMemoryBufferWithSize & operator=(ScopedMemoryBufferWithSize && other)
215 {
216 if (this != &other)
217 {
218 mCount = other.mCount;
219 other.mCount = 0;
220 }
221 ScopedMemoryBuffer<T>::operator=(std::move(other));
222 return *this;
223 }
224
225 // return the size as count of elements
226 inline size_t AllocatedSize() const { return mCount; }
227
228 chip::Span<T> Span() { return chip::Span<T>(this->Get(), AllocatedSize()); }
229 chip::Span<const T> Span() const { return chip::Span<T>(this->Get(), AllocatedSize()); }
230
231 void Free()
232 {
233 mCount = 0;
234 ScopedMemoryBuffer<T>::Free();
235 }
236
242 CHECK_RETURN_VALUE T * Release()
243 {
244 T * buffer = ScopedMemoryBuffer<T>::Release();
245 mCount = 0;
246 return buffer;
247 }
248
249 ScopedMemoryBufferWithSize & Calloc(size_t elementCount)
250 {
251 ScopedMemoryBuffer<T>::Calloc(elementCount);
252 if (this->Get() != nullptr)
253 {
254 mCount = elementCount;
255 }
256 return *this;
257 }
258
259 ScopedMemoryBufferWithSize & Alloc(size_t elementCount)
260 {
261 ScopedMemoryBuffer<T>::Alloc(elementCount);
262 if (this->Get() != nullptr)
263 {
264 mCount = elementCount;
265 }
266 return *this;
267 }
268
269 // Allow creating ScopedMemoryBufferWithSize from Span, so we
270 // don't have to reinvent it in various places.
271 template <class U, typename = std::enable_if_t<sizeof(U) == sizeof(T) && std::is_convertible_v<U *, T *>>>
272 ScopedMemoryBufferWithSize & CopyFromSpan(const chip::Span<const U> & span)
273 {
274 static_assert(std::is_trivially_copyable_v<U>, "Span<const U> must be trivially copyable");
275
276 if (span.size() == 0)
277 {
278 Free();
279 return *this;
280 }
281
282 ScopedMemoryBufferWithSize<T>::Alloc(span.size());
283
284 if (AllocatedSize() > 0)
285 {
286 memcpy(ScopedMemoryBuffer<T>::Get(), span.data(), AllocatedSize());
287 }
288
289 return *this;
290 }
291
292private:
293 size_t mCount = 0;
294};
