82// These are MEIs, 0xFFFF is not a valid manufacturer code,

92static constexpr uint16_t ExtractIdFromMEI(uint32_t aMEI)
93{
94 constexpr uint32_t kIdMask = 0x0000'FFFF;
95 return static_cast<uint16_t>(aMEI & kIdMask);
96}
97
98static constexpr uint16_t ExtractVendorFromMEI(uint32_t aMEI)
99{
100 constexpr uint32_t kVendorMask = 0xFFFF'0000;
101 constexpr uint32_t kVendorShift = 16;
102 return static_cast<uint16_t>((aMEI & kVendorMask) >> kVendorShift);
103}
104
105static constexpr bool IsValidVendorId(uint16_t aVendorId)
106{
107 return aVendorId <= kMaxVendorId;
108}

110constexpr bool IsValidClusterId(ClusterId aClusterId)
111{
112 const auto id = ExtractIdFromMEI(aClusterId);
113 const auto vendor = ExtractVendorFromMEI(aClusterId);
114 // Cluster id suffixes in the range 0x0000 to 0x7FFF indicate a standard
115 // cluster.
116 //
117 // Cluster id suffixes in the range 0xFC00 to 0xFFFE indicate an MS cluster.
118 return IsValidVendorId(vendor) && ((vendor == 0x0000 && id <= 0x7FFF) || (vendor >= 0x0001 && id >= 0xFC00 && id <= 0xFFFE));
119}
120
121constexpr bool IsGlobalAttribute(AttributeId aAttributeId)
122{
123 const auto id = ExtractIdFromMEI(aAttributeId);
124 const auto vendor = ExtractVendorFromMEI(aAttributeId);
125 // Attribute id suffixes in the range 0xF000 to 0xFFFE indicate a standard
126 // global attribute.
127 return (vendor == 0x0000 && id >= 0xF000 && id <= 0xFFFE);
128}
129
130constexpr bool IsValidAttributeId(AttributeId aAttributeId)
131{
132 const auto id = ExtractIdFromMEI(aAttributeId);
133 const auto vendor = ExtractVendorFromMEI(aAttributeId);
134 // Attribute id suffixes in the range 0x0000 to 0x4FFF indicate a non-global
135 // attribute (standard or MS). The vendor id must not be wildcard in this
136 // case.
137 return (id <= 0x4FFF && IsValidVendorId(vendor)) || IsGlobalAttribute(aAttributeId);
138}
139
140constexpr bool IsValidCommandId(CommandId aCommandId)
141{
142 const auto id = ExtractIdFromMEI(aCommandId);
143 const auto vendor = ExtractVendorFromMEI(aCommandId);
144
145 // Command id suffixes must be in the range 0x00 to 0xFF, and the prefix
146 // must be a valid prefix (standard or MC).
147 return id <= 0xFF && IsValidVendorId(vendor);
148}

150constexpr bool IsValidDeviceTypeId(DeviceTypeId aDeviceTypeId)
151{
152 const DeviceTypeId kIdMask = 0x0000'FFFF;
153 const DeviceTypeId kVendorMask = 0xFFFF'0000;
154 const auto id = aDeviceTypeId & kIdMask;
155 const auto vendor = aDeviceTypeId & kVendorMask;
156 return vendor <= 0xFFFE'0000 && id <= 0xBFFF;
157}

159constexpr bool IsValidEndpointId(EndpointId aEndpointId)
160{
161 return aEndpointId != kInvalidEndpointId;
162}

164constexpr bool IsValidFabricIndex(FabricIndex fabricIndex)
165{
166 return (fabricIndex >= kMinValidFabricIndex) && (fabricIndex <= kMaxValidFabricIndex);
167}
168
169constexpr bool IsValidFabricId(FabricId fabricId)
170{
171 return fabricId != kUndefinedFabricId;
172}
