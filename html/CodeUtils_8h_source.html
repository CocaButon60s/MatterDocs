795template <typename T>
796constexpr bool ArrayIsSorted(const T * aArray, size_t aLength)
797{
798 if (aLength == 0 || aLength == 1)
799 {
800 return true;
801 }
802
803 if (aArray[0] > aArray[1])
804 {
805 return false;
806 }
807
808 return ArrayIsSorted(aArray + 1, aLength - 1);
809}
810
811template <typename T, size_t N>
812constexpr bool ArrayIsSorted(const T (&aArray)[N])
813{
814 return ArrayIsSorted(aArray, N);
815}
