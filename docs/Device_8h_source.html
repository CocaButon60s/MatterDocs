64 inline std::string GetLocation() { return mLocation; };
65 inline std::string GetZone() { return mZone; };
66 inline void SetZone(std::string zone) { mZone = zone; };
67
68private:
69 virtual void HandleDeviceChange(Device * device, Device::Changed_t changeMask) = 0;
