73struct EndpointListStorage : public Structs::EndpointListStruct::Type
74{
75 EndpointListStorage(){};
76
77 EndpointListStorage(uint16_t aEpListId, const CharSpan & aEpListName, EndpointListTypeEnum aEpListType,
78 const DataModel::List<const EndpointId> & aEndpointList)
79 {
80 Set(aEpListId, aEpListName, aEpListType, aEndpointList);
81 }
82
83 EndpointListStorage(const EndpointListStorage & aEpList) { *this = aEpList; }
84
85 EndpointListStorage & operator=(const EndpointListStorage & aEpList)
86 {
87 Set(aEpList.endpointListID, aEpList.name, aEpList.type, aEpList.endpoints);
88 return *this;
89 }
90
91 void Set(uint16_t aEpListId, const CharSpan & aEpListName, EndpointListTypeEnum aEpListType,
92 const DataModel::List<const EndpointId> & aEndpointList)
93 {
94 endpointListID = aEpListId;
95 type = aEpListType;
96 size_t epListSize = std::min(aEndpointList.size(), MATTER_ARRAY_SIZE(mEpList));
97
98 for (size_t index = 0; index < epListSize; index++)
99 {
100 mEpList[index] = aEndpointList[index];
101 }
102 endpoints = DataModel::List<const EndpointId>(Span(mEpList, epListSize));
103 MutableCharSpan epListName(mBuffer);
104 CopyCharSpanToMutableCharSpanWithTruncation(aEpListName, epListName);
105 name = epListName;
106 }
107
108private:
109 char mBuffer[kEndpointListNameMaxSize];
110 EndpointId mEpList[kEndpointListMaxSize];
111};
