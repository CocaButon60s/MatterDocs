256 CHIP_ERROR InitializeStaticResourcesBeforeServerInit()
257 {
258 // KVS-based persistent storage delegate injection
259 if (persistentStorageDelegate == nullptr)
260 {
261 chip::DeviceLayer::PersistedStorage::KeyValueStoreManager & kvsManager =
262 DeviceLayer::PersistedStorage::KeyValueStoreMgr();
263 ReturnErrorOnFailure(sKvsPersistenStorageDelegate.Init(&kvsManager));
264 this->persistentStorageDelegate = &sKvsPersistenStorageDelegate;
265 }
266
267 // PersistentStorageDelegate "software-based" operational key access injection
268 if (this->operationalKeystore == nullptr)
269 {
270 // WARNING: PersistentStorageOperationalKeystore::Finish() is never called. It's fine for
271 // for examples and for now.
272 ReturnErrorOnFailure(sPersistentStorageOperationalKeystore.Init(this->persistentStorageDelegate));
273 this->operationalKeystore = &sPersistentStorageOperationalKeystore;
274 }
275
276 // OpCertStore can be injected but default to persistent storage default
277 // for simplicity of the examples.
278 if (this->opCertStore == nullptr)
279 {
280 // WARNING: PersistentStorageOpCertStore::Finish() is never called. It's fine for
281 // for examples and for now, since all storage is immediate for that impl.
282 ReturnErrorOnFailure(sPersistentStorageOpCertStore.Init(this->persistentStorageDelegate));
283 this->opCertStore = &sPersistentStorageOpCertStore;
284 }
285
286 // Injection of report scheduler WILL lead to two schedulers being allocated. As recommended above, this should only be used
287 // for IN-TREE examples. If a default scheduler is desired, the basic ServerInitParams should be used by the application and
288 // CommonCaseDeviceServerInitParams should not be allocated.
289 if (this->reportScheduler == nullptr)
290 {
291 reportScheduler = &sReportScheduler;
292 }
293
294 // Session Keystore injection
295 this->sessionKeystore = &sSessionKeystore;
296
297 // Group Data provider injection
298 sGroupDataProvider.SetStorageDelegate(this->persistentStorageDelegate);
299 sGroupDataProvider.SetSessionKeystore(this->sessionKeystore);
300 ReturnErrorOnFailure(sGroupDataProvider.Init());
301 this->groupDataProvider = &sGroupDataProvider;
302
303#if CHIP_CONFIG_ENABLE_SESSION_RESUMPTION
304 ReturnErrorOnFailure(sSessionResumptionStorage.Init(this->persistentStorageDelegate));
305 this->sessionResumptionStorage = &sSessionResumptionStorage;
306#else
307 this->sessionResumptionStorage = nullptr;
308#endif
309
310 // Inject access control delegate
311 this->accessDelegate = Access::Examples::GetAccessControlDelegate();
312
313 // Inject ACL storage. (Don't initialize it.)
314 this->aclStorage = &sAclStorage;
315
316#if CHIP_CONFIG_PERSIST_SUBSCRIPTIONS
317 ChipLogProgress(AppServer, "Initializing subscription resumption storage...");
318 ReturnErrorOnFailure(sSubscriptionResumptionStorage.Init(this->persistentStorageDelegate));
319 this->subscriptionResumptionStorage = &sSubscriptionResumptionStorage;
320#else
321 ChipLogProgress(AppServer, "Subscription persistence not supported");
322#endif
323
324#if CHIP_CONFIG_ENABLE_ICD_CIP
325 if (this->icdCheckInBackOffStrategy == nullptr)
326 {
327 this->icdCheckInBackOffStrategy = &sDefaultICDCheckInBackOffStrategy;
328 }
329#endif
330
331 return CHIP_NO_ERROR;
332 }
333
334private:
335 static KvsPersistentStorageDelegate sKvsPersistenStorageDelegate;

337 static Credentials::PersistentStorageOpCertStore sPersistentStorageOpCertStore;
338 static Credentials::GroupDataProviderImpl sGroupDataProvider;

504 void OnGroupAdded(chip::FabricIndex fabric_index, const Credentials::GroupDataProvider::GroupInfo & new_group) override
505 {
506 const FabricInfo * fabric = mServer->GetFabricTable().FindFabricWithIndex(fabric_index);

526 return;
527 }
528
529 mServer->GetTransportManager().MulticastGroupJoinLeave(
530 Transport::PeerAddress::Multicast(fabric->GetFabricId(), old_group.group_id), false);
531 };

537 class ServerFabricDelegate final : public chip::FabricTable::Delegate
538 {
539 public:
540 ServerFabricDelegate() {}
541
542 CHIP_ERROR Init(Server * server)
543 {
544 VerifyOrReturnError(server != nullptr, CHIP_ERROR_INVALID_ARGUMENT);
545
546 mServer = server;
547 return CHIP_NO_ERROR;
548 }
549
550 void OnFabricRemoved(const FabricTable & fabricTable, FabricIndex fabricIndex) override
551 {
552 (void) fabricTable;
553 ClearCASEResumptionStateOnFabricChange(fabricIndex);
554 ClearSubscriptionResumptionStateOnFabricChange(fabricIndex);
555
556 Credentials::GroupDataProvider * groupDataProvider = mServer->GetGroupDataProvider();
557 if (groupDataProvider != nullptr)
558 {
559 CHIP_ERROR err = groupDataProvider->RemoveFabric(fabricIndex);
560 if (err != CHIP_NO_ERROR)
561 {
562 ChipLogError(AppServer,
563 "Warning, failed to delete GroupDataProvider state for fabric index 0x%x: %" CHIP_ERROR_FORMAT,
564 static_cast<unsigned>(fabricIndex), err.Format());
565 }
566 }
567
568 // Remove access control entries in reverse order (it could be any order, but reverse order
569 // will cause less churn in persistent storage).
570 CHIP_ERROR aclErr = Access::GetAccessControl().DeleteAllEntriesForFabric(fabricIndex);
571 if (aclErr != CHIP_NO_ERROR)
572 {
573 ChipLogError(AppServer, "Warning, failed to delete access control state for fabric index 0x%x: %" CHIP_ERROR_FORMAT,
574 static_cast<unsigned>(fabricIndex), aclErr.Format());
575 }
576
577 // Remove ACL extension entry for the given fabricIndex.
578 auto & storage = mServer->GetPersistentStorage();
579 aclErr = storage.SyncDeleteKeyValue(DefaultStorageKeyAllocator::AccessControlExtensionEntry(fabricIndex).KeyName());
580
581 if (aclErr != CHIP_NO_ERROR && aclErr != CHIP_ERROR_PERSISTED_STORAGE_VALUE_NOT_FOUND)
582 {
583 ChipLogError(AppServer, "Warning, failed to delete ACL extension entry for fabric index 0x%x: %" CHIP_ERROR_FORMAT,
584 static_cast<unsigned>(fabricIndex), aclErr.Format());
585 }
586
587 mServer->GetCommissioningWindowManager().OnFabricRemoved(fabricIndex);
588 }
589
590 void OnFabricUpdated(const FabricTable & fabricTable, chip::FabricIndex fabricIndex) override
591 {
592 (void) fabricTable;
593 ClearCASEResumptionStateOnFabricChange(fabricIndex);
594 }
595
596 private:
597 void ClearCASEResumptionStateOnFabricChange(chip::FabricIndex fabricIndex)
598 {
599 auto * sessionResumptionStorage = mServer->GetSessionResumptionStorage();
600 VerifyOrReturn(sessionResumptionStorage != nullptr);
601 CHIP_ERROR err = sessionResumptionStorage->DeleteAll(fabricIndex);
602 if (err != CHIP_NO_ERROR)
603 {
604 ChipLogError(AppServer,
605 "Warning, failed to delete session resumption state for fabric index 0x%x: %" CHIP_ERROR_FORMAT,
606 static_cast<unsigned>(fabricIndex), err.Format());
607 }
608 }
609

618 "Warning, failed to delete subscription resumption state for fabric index 0x%x: %" CHIP_ERROR_FORMAT,
619 static_cast<unsigned>(fabricIndex), err.Format());
620 }
621 }

633 class IgnoreRootExpirationValidityPolicy : public Credentials::CertificateValidityPolicy
634 {
635 public:
636 IgnoreRootExpirationValidityPolicy() {}
637
638 void Init(Credentials::CertificateValidityPolicy * providedPolicy) { mProvidedPolicy = providedPolicy; }
639
640 CHIP_ERROR ApplyCertificateValidityPolicy(const Credentials::ChipCertificateData * cert, uint8_t depth,
641 Credentials::CertificateValidityResult result) override
642 {
643 switch (result)
644 {
645 case Credentials::CertificateValidityResult::kExpired:
646 case Credentials::CertificateValidityResult::kExpiredAtLastKnownGoodTime:
647 case Credentials::CertificateValidityResult::kTimeUnknown: {
648 Credentials::CertType certType;
649 ReturnErrorOnFailure(cert->mSubjectDN.GetCertType(certType));
650 if (certType == Credentials::CertType::kRoot)
651 {
652 return CHIP_NO_ERROR;
653 }
654
655 break;
656 }
657 default:
658 break;
659 }
660
661 if (mProvidedPolicy)
662 {
663 return mProvidedPolicy->ApplyCertificateValidityPolicy(cert, depth, result);
664 }
665
666 return Credentials::CertificateValidityPolicy::ApplyDefaultPolicy(cert, depth, result);
667 }
668
669 private:
670 Credentials::CertificateValidityPolicy * mProvidedPolicy = nullptr;
671 };
